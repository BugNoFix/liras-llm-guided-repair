NurseAssistant

agents:
    humans: Nurse, Doctor
    robots: MP, M, MM
locations: radiotherapy, reservedRoom1, reservedRoom2
resources: towel, item2

Nurse:
    1: receiveFrom MP
    5: receiveFrom MM

Doctor:

M:
    1: stop 
    restart if !possess(Nurse, towel)
    3: skip if !possess(Nurse, towel)
       fetch towel
    4: skip if possess(Doctor, item2)
       fetch item2

MP:
    1: moveTo radiotherapy until possess(Nurse, towel)
    2: moveTo M
    3-4: stop
    5: moveTo Doctor

MM: 
    1-5: moveTo reservedRoom1
    moveTo Nurse
    deliver towel
